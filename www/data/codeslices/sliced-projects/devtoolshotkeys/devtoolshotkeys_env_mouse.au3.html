  Func GlobalEnvMouse()
    Opt("MouseCoordMode",1)
    Local $pos=MouseGetPos();
    Local $mouse_x=$pos[0];
    Local $mouse_y=$pos[1];
    GlobalEnvItem('pos',$pos)
    GlobalEnvItem('mouse_x',$mouse_x)
    GlobalEnvItem('mouse_y',$mouse_y)
  EndFunc
