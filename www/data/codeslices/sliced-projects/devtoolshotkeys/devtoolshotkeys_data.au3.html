Func Of($what,$d); _x _y _w _h _tabs
  Local $r=Eval($type&'_'&$name&'_'&$what)
  If @Error Then
    Return $d
  EndIf
  Return $r
EndFunc

Func OfType($what,$d); _x _y _w _h _tabs
  Local $r=Eval($type&'_'&$what)
  If @Error Then
    Return $d
  EndIf
  Return $r
EndFunc


Func QuadrantValFromWindow($q)
  Return QuadrantValFromRect($q,$w_p)
EndFunc

Func QuadrantValFromRect($q,$r)
  ; xywh
  Local $a[2]
  If $q='tl' Then
    $a[0]=$r[0]
    $a[1]=$r[1]
  EndIf
  If $q='tr' Then
    $a[0]=$r[0]+$r[2]
    $a[1]=$r[1]
  EndIf
  If $q='bl' Then
    $a[0]=$r[0]
    $a[1]=$r[1]+$r[3]
  EndIf
  If $q='br' Then
    $a[0]=$r[0]+$r[2]
    $a[1]=$r[1]+$r[3]
  EndIf
  Return $a;
EndFunc

Func QuadrantValOfType($t=Default)
  Local $q=Quadrant($t)
  Return QuadrantValFromWindow($q)
EndFunc

Func Quadrant($t=Default,$n=Default)
  If $t=Default Then $t=$type
  If $n=Default Then $n=$name
  Local $vn;=$t&'_'&$n&'_q';
  If $n<>'' Then
    $vn=$t&'_'&$n&'_q';
  Else
    $vn=$t&'_q';
  EndIf
  If IsDeclared($vn) Then
    Return Eval($vn)
  EndIf
  Return 'tl';
EndFunc

Func OriginRet($x,$y,$xy=Default)
  Local $a[2]
  $a[0]=$x
  $a[1]=$y
  If $xy=Default Then
    Return $a;
  EndIf
  If $xy='x' Then Return $x
  Return $y
EndFunc

Func Origin($xy=Default,$t=Default,$n=Default)
  If $t=Default Then $t=$type
  If $n=Default Then $n=$name
  Local $vn
  If $n<>'' Then
    $vn=$t&'_'&$n&'_o';
  Else
    $vn=$t&'_o';
  EndIf
  If $xy=Default Then
    $vn=$vn&'_a';
  Else
    $vn=$vn&'_'&$xy;
  EndIf
  If IsDeclared($vn) Then
    Return Eval($vn)
  EndIf
  Return OriginRet(0,0,$xy)
EndFunc
Func OriginOfType($xy=Default,$t=Default)
  ; quadrant origin
  ;  has      has     wq + or
  ;  has      not     wq + 0 0
  ;  not      has     0 0 + or
  If $t=Default Then $t=$type
  Local $qv=QuadrantValOfType($t)
  Local $ov=Origin(Default,$t,'')
  Local $a[2]
  Local $x=$qv[0]+$ov[0]
  Local $y=$qv[1]+$ov[1]
  Return OriginRet($x,$y,$xy)
EndFunc

Func BothOrNeitherExit($s,$t=Default,$n=Default)
  If $t=Default And $n<>Default Then
    ErrorExit("BothOrNeitherExit:"&$s)
  EndIf
  If $t<>Default And $n=Default Then
    ErrorExit("BothOrNeitherExit:"&$s)
  EndIf
EndFunc

Func OriginOf($xy=Default,$t=Default,$n=Default)
  ; must call with t and n or not t and n
  BothOrNeitherExit("OriginOf",$t,$n)
  If $t=Default Then $t=$type
  If $n=Default Then $n=$name
  Local $ot=OriginOfType(Default,$t)
  Local $on=Origin(Default,$t,$n)
  Local $x=$ot[0]+$on[0]
  Local $y=$ot[1]+$on[1]
  Return OriginRet($x,$y,$xy)
EndFunc
Func DimExistsForType($d,$t=Default)
  Return DimExists($d,$t,'')
EndFunc
Func DimExists($d,$t=Default,$n=Default)
  ;VarExists()
  If $t=Default Then $t=$type
  If $n=Default Then $n=$name
  Local $vn
  If $n<>'' Then
    $vn=$t&'_'&$n&'_'&$d;
  Else
    $vn=$t&'_'&$d;
  EndIf
  Return IsDeclared($vn)
EndFunc
Func DimOf($d,$t=Default,$n=Default)
  ; must exist, both or neither
  ; use with DimExists
  BothOrNeitherExit("DimOf",$t,$n)
  If $t=Default Then $t=$type
  If $n=Default Then $n=$name
  Local $vn
  If $n<>'' Then
    $vn=$t&'_'&$n&'_'&$d;
  Else
    $vn=$t&'_'&$d;
  EndIf
  Return Eval($vn)
EndFunc
Func DimAlias($s)
  If $s='l' Then Return 'x';
  If $s='x' Then Return 'l';
  If $s='y' Then Return 't';
  If $s='t' Then Return 'y';
  Return ''
EndFunc
Func DimAliasHas($s)
  Return DimAlias($s)<>''
EndFunc

Func DimForName($d,$t=Default,$n=Default)
  ; if not exist offer parent xy center
  BothOrNeitherExit("DimForName",$t,$n)
  If $t=Default Then $t=$type
  If $n=Default Then $n=$name
  Local $bug=True
  $bug=False
  If DimAliasHas($d) Then
    If DimExists($d,$t,$n) Then
      If $bug Then MsgBox(0,'DimForName '&$d&' '&$t&'_'&$n,'Exists1 normal');
      Return DimOf($d,$t,$n)
    EndIf
    If DimExists(DimAlias($d),$t,$n) Then
      If $bug Then MsgBox(0,'DimForName '&$d&' '&$t&'_'&$n,'Exists2 alias');
      Return DimOf(DimAlias($d),$t,$n)
    EndIf
  Else
    If DimExists($d,$t,$n) Then
      If $bug Then MsgBox(0,'DimForName '&$d&' '&$t&'_'&$n,'Exists3 normal no alias');
      Return DimOf($d,$t,$n)
    EndIf
  EndIf
  If $d='y' Or $d='t' Then
    If DimExistsForType('h',$t) Then Return Int(DimOf('h',$t,'')/2)
    If $bug Then MsgBox(0,'DimForName '&$d&' '&$t&'_'&$n,'No height for type');
  EndIf
  If $d='l' Or $d='x' Then
    If DimExistsForType('w',$t) Then Return Int(DimOf('w',$t,'')/2)
    If $bug Then MsgBox(0,'DimForName '&$d&' '&$t&'_'&$n,'No Width for type');
  EndIf
  Return 0
EndFunc

Func DimForType($d,$t=Default)
  ; get raw
  If $t=Default Then $t=$type
  Local $n=''
  If DimAliasHas($d) Then
    If DimExistsForType($d,$t) Then
      Return DimOf($d,$t,$n)
    EndIf
    If DimExistsForType(DimAlias($d),$t) Then
      Return DimOf(DimAlias($d),$t,$n)
    EndIf
  Else
    If DimExistsForType($d,$t) Then
      Return DimOf($d,$t,$n)
    EndIf
  EndIf
  Return 0
EndFunc



Func Position($xy=Default,$t=Default,$n=Default)
  BothOrNeitherExit("Position",$t,$n)
  If $t=Default Then $t=$type
  If $n=Default Then $n=$name
  Local $bug=True;
  $bug=False
  $bug=$__positionbug;
  If $__noparents Then $bug=False
  Local $s='';
  Local $ot=OriginOfType(Default,$t);
  $s=ListAppend($s,'TypeO $ot:'&Stringify($ot),@CRLF)
  Local $ty=DimForType('y',$t)
  Local $tx=DimForType('x',$t)
  $s=ListAppend($s,'Type: $tx $ty:'&$tx&' '&$ty,@CRLF)
  If $n='' Then
    Local $xx=$ot[0]+$tx;
    Local $yy=$ot[1]+$ty;
    $s=ListAppend($s,'Final:$xx $yy:'&$xx&' '&$yy,@CRLF)
    ;If $bug Then MsgBox(0,'Position:'&$t&' '&$n,$s)
    If $bug Then ClipPut('Position:'&$t&' '&$n&@CRLF&$s)
    Return OriginRet($xx,$yy,$xy)
  EndIf
  Local $on=Origin(Default,$t,$n);
  $s=ListAppend($s,'NameO $on:'&Stringify($on),@CRLF)
  Local $x=DimForName('x',$t,$n)
  Local $y=DimForName('y',$t,$n)
  $s=ListAppend($s,'Name:$x $y:'&$x&' '&$y,@CRLF)
  Local $xx=$x+$ot[0]+$on[0]+$tx;
  Local $yy=$y+$ot[1]+$on[1]+$ty;
  $s=ListAppend($s,'Final:$xx $yy:'&$xx&' '&$yy,@CRLF)
  ;If $bug Then MsgBox(0,'Position:'&$t&' '&$n,$s)
  If $bug Then ClipPut('Position:'&$t&' '&$n&@CRLF&$s)
  Return OriginRet($xx,$yy,$xy)
EndFunc

Func IsCtrl($t=Default,$n=Default)
  BothOrNeitherExit("IsCtrl",$t,$n)
  If $t=Default Then $t=$type
  If $n=Default Then $n=$name
  If Not IsDeclared($t&'_ctrl_a')=1 Then
    ;ErrorExit("IsCtrl, not found:"&$t&'_ctrl_a');
    Return False
  EndIf
  Local $v=Eval($t&'_ctrl_a');
  Return ArrayHas($v,$n)
EndFunc

Func Parent($t=Default,$n=Default)
  BothOrNeitherExit("Parent",$t,$n)
  If $t=Default Then $t=$type
  If $n=Default Then $n=$name
  If IsDeclared($t&'_parent')=1 Then
    Return Eval($t&'_parent')
  EndIf
  Return ''
EndFunc

Func IsPaneScroll($t=Default,$n=Default)
  BothOrNeitherExit("IsPaneScroll",$t,$n)
  If $t=Default Then $t=$type
  If $n=Default Then $n=$name
  If StringLen($t)<4 Then
    Return False;
  EndIf
  If StringLen($n)<5 Then
    Return False;
  EndIf
  If StringRight($t,4)<>'Pane' Then
    Return False;
  EndIf
  If StringLeft($n,6)<>'Scroll' Then
    Return False;
  EndIf
  Return True;
EndFunc
