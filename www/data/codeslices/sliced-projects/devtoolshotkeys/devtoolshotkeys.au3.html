Func DoTypeName()
  ;ToolTip("HERE",$left,$ConsoleTabs_y+100)
  ;Sleep(10000)
  If Not $__noparents Then
    DoParents()
  EndIf
  If IsPaneScroll() Then
    DoPaneScroll()
    Util_mousehome()
    Return;
  EndIf
  Local $n=$name;
  If StringLeft($n,6)=="Toggle" Then
    $n=StringMid($n,7)
  Else
    If StringLeft($n,5)=="Focus" Then
      $n=StringMid($n,6)
    EndIf
  EndIf
  Local $p=Position(Default,$type,$n)
  Local $x=$p[0];Position('x')
  Local $y=$p[1];Position('y')
  ;if name has x or y then click
  ;if name is in _ctrl section then press ctrl
  ;if name is in _rightclick section then rightclick
  ;if name is in _middleclick section then rightclick
  ;if name is Selector or ends with Selector then doubleclick
  ;if type is pane then names scrolldown, scrollup scrollpagedown, scrollpageup, scrollhome, scrollend
  Local $doit=False;
  $doit=True
  If $doit Then
    ;Msg('$x:'&$x&' $y:'&$y)
    MouseMove($x,$y,0)
    If True Then
      ;Msg('clicl')
      If IsCtrl() Then
        Send('{Ctrldown}');
      EndIf
      ToolTip("")
      Sleep(250)
      MouseClick('left',$x,$y);
      Sleep(250)
      If IsCtrl() Then
        Send('{Ctrlup}');
      EndIf
      ToolTip($type&' '&$name&' '&$x&' '&$y,$x,$y)
      Util_mousehome()
    EndIf
  EndIf
  If $__noparents Then
    Sleep(250)
  Else
    Sleep(3000)
  EndIf
EndFunc

Func DoPaneScroll($t=Default,$n=Default)
  BothOrNeitherExit("Position",$t,$n)
  If $t=Default Then $t=$type
  If $n=Default Then $n=$name
  ;If Not DimExists('x',$t,$n) Then Return -1;
  ;If Not DimExists('y',$t,$n) Then Return -1;
  Local $p=Position(Default,$t,'')
  Local $dir='up';
  Local $amt=1;
  If $n='ScrollHome' Or $n='ScrollEnd' Then $amt=1000
  If $n='ScrollPageDown' Or $n='ScrollPageUp' Then $amt=3
  If $n='ScrollDown' Or $n='ScrollPageDown' Or $n='ScrollEnd' Then $dir='down'
  ;Msg('p '&$p[0]&' '&$p[1])
  ;Msg('amt '&$amt)
  MouseMove($p[0]+5,$p[1]+5,0)
  MouseWheel($dir,$amt);
  Return 0
EndFunc

Func Dispatch()
  If $type=='cmd' Then
    TryCmd()
  Else
    DoTypeName()
  EndIf
EndFunc

Func DoParents($t=Default,$n=Default)
  BothOrNeitherExit("DoParents",$t,$n)
  If $t=Default Then $t=$type
  If $n=Default Then $n=$name
  ;Msg('DoParents:t:'&$t&' n:'&$n)
  Local $a=Parent($t,$n);
  If $a='' Then Return
  ;Msg('a:"'&$a&'"')
  Local $b=ToList($a,':')
  DoParents($b[0],$b[1])
  Local $c=ToList($a,':');
  $__noparents=True
  DoThis($c[0],$c[1])
  $__noparents=False
EndFunc

Func DoThis($t,$n)
  Local $wasdoing=$__selfcall
  Local $t0=$type
  Local $n0=$name
  $__selfcall=True;
  $type=$t;
  $name=$n;
  ;ValidateType()
  ;ValidateName()
  Dispatch()
  $name=$n0
  $type=$t0
  $__selfcall=$wasdoing;
EndFunc

Func TryCmd()
  Local $a=IniReadSection($f_in,$name&'_cmd')
  If Not IsArray($a) Then
    MsgBox(0,$thisfile&' Command Section Not Found, Exiting',$name&'_cmd')
    Exit
  EndIf
  Local $dec=IniRead($f_in,$name&'_cmd','declare','');
  Local $__x
  Local $__v
  If $dec=='' Then
    ;MsgBox(0,$thisfile&' Command declare Not Found, Exiting',$name&'_cmd')
    ;Exit
  Else
    $dec=StringSplit($dec,' ',2)
    For $__x=0 To UBound($dec)-1
      $__v=IniRead($f_in,$name&'_cmd',$dec[$__x],'');
      $__v=Execute($__v)
      Assign($dec[$__x],$__v,1);local
    Next
  EndIf
  ;MsgBox(0,'vals:x,w',$x&' '&$w)
  $__x=0
  While True
    $__x=$__x+1
    $__v=IniRead($f_in,$name&'_cmd',''&$__x,'');
    If $__v=='' Then
      ExitLoop
    EndIf
    ;Msg($__v)
    $__v=Execute($__v)
  WEnd
  Util_mousehome()
EndFunc

Func CheckValid($nx,$ny)
  If $nx<$left Or $nx>$right Or $ny<$top Or $ny>$bottom Then
    ToolTip("Out of Bounds:"&$type&' '&$name);
    Return False;
  EndIf
  Return True;
EndFunc

Func AfterKeys()
  Local $v=IniRead($f_in,$type&'_'&$name,'afterkeys','');
  If $v<>'' Then
    Sleep(500)
    Send($v)
  EndIf
EndFunc

Func Util_mousehome()
  MouseMove($mouse_x,$mouse_y,0);
EndFunc
