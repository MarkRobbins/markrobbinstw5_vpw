  Func GlobalEnvWin()
    Local $wh1='';
    If FileExists($f_ini) Then
      $wh1=IniRead($f_ini,'data','hwnd','')
    EndIf
    ;Msg($wh1)
    Local $wh=WinGetHandle('[active]')
    Local $wh0=WinGetHandle('[active]')
    If $wh1<>'' Then
      $wh=HWnd($wh1)
    EndIf
    If Not WinExists($wh) Then
      ;1 = Match the title from the start (default)
      Opt("WinTitleMatchMode",1)
      $wh=WinGetHandle('Developer Tools -');
    EndIf
    Local $w_p=WinGetPos($wh)
    GlobalEnvItem('wh',$wh)
    GlobalEnvItem('w_p',$w_p)
    GlobalEnvItem('left',$w_p[0])
    GlobalEnvItem('top',$w_p[1])
    GlobalEnvItem('right' ,$w_p[0]+$w_p[2])
    GlobalEnvItem('bottom',$w_p[1]+$w_p[3])
    GlobalEnvItem('width' ,$w_p[2])
    GlobalEnvItem('height',$w_p[3])
    GlobalEnvItem('title',WinGetTitle($wh))
  EndFunc
