  Func IniSectionTo($sec,$f,$nf,$vf,$k,$quiet)
    Local $a=GE_IniSectionRead($sec);
    If Not IsArray($a) Then
      If $quiet Then
        Return
      Else
        ErrorExit("Cannot find section '"&$sec&"'")
      EndIf
    EndIf
    If $sec='_at2' Then
      ;_ArrayDisplay($a)
    EndIf
    GE_IniItemsTo($a,$f,$nf,$vf,$k)
  EndFunc
  Func GE_IniItemsTo($a,$f,$nf,$vf,$k)
    Local $x,$end=$a[0][0]
    Local $is_list=GE_IsSectionBase('list')
    $is_list=Iif($is_list,'_a','')
    For $x=1 To $end
      Local $na=GE_AsWho($a[$x][0],$nf);
      Local $va=GE_AsWhat($a[$x][1],$vf); might return [n][v]
      Local $isval=True
      If IsArray($va) And UBound($va,0)=2 Then
        $isval=False
      EndIf
      If $isval Then
        Local $y,$endy=EndOf($na)
        For $y=0 To $endy
          Call($f,$na[$y]&$is_list,$va,$k)
        Next
      Else
        GE_IniValsTo2($f,$na,$va,$k)
      EndIf
    Next
  EndFunc
  Func GE_IniValsTo2($f,$na,$va,$k)
    Local $x,$endx=EndOf($na)
    Local $is_list=GE_IsSectionBase('list')
    $is_list=Iif($is_list,'_a','')
    For $x=0 To $endx
      Local $n=$na[$x]
      Local $y,$endy=EndOf($va)
      For $y=0 To $endy
        Local $sfx=$va[$y][0]
        Local $vv=$va[$y][1]
        Call($f,$n&'_'&$sfx&$is_list,$vv,$k)
      Next
    Next
  EndFunc
  Func GE_SectionWhatType($sec)
    Return Iif(ArrayHas($__base_exec,$sec),'Execute','Literal')
  EndFunc
  Func GE_SectionWhoType($sec)
    If ArrayHas($__base_pfx,$sec) Then Return 'Pfx'
    Local $a=ToList($sec)
    If UBound($a)<>2 Then Return 'Many'
    Return 'Lace';
  EndFunc
  Func GE_ReadBases()
    Local $a=$__base_read;
    Local $hit_exec=False;
    Local $x,$end=EndOf($a)
    For $x=0 To $end
      Local $sec=$a[$x]
      Local $nf=GE_SectionWhoType($sec)
      If $hit_exec=False Then
        If ArrayHas($__base_exec,$sec) Then
          $hit_exec=True
          GlobalEnvWin()
          GlobalEnvMouse()
        EndIf
      EndIf
      Local $quiet=GE_IsSectionQuiet($sec)
      Local $vf=GE_SectionWhatType($sec);
      If $sec='_o' Then
        ;MsgBox(0,$sec&'- nf vf',$nf&' '&$vf)
      EndIf
      IniSectionTo($sec,'DoAssign',$nf,$vf,'_'&$sec,$quiet)
      If $sec='_types' Then
        GE_DoTypes()
      EndIf
    Next
  EndFunc
  Func GE_DoLaceSection($sec)
    Local $nv=ToList($sec,' ')
    Local $type=$nv[0]
    Local $sfx=$nv[1]
    IniSectionTo($sec,'DoAssign','Lace','Execute','__type_'&$type,False)
  EndFunc
  Func GE_DoType($n)
    Local $a=$__base_dims;
    Local $b=IniReadSectionNames($f_in)
    Local $x,$end=EndOf($a)
    For $x=0 To $end
      Local $nn=$n&' '&$a[$x];
      If ArrayHas($b,$nn) Then
        GE_DoLaceSection($nn)
      EndIf
    Next
  EndFunc
  Func GE_DoTypes()
    Local $a=IniReadSection($f_in,'_types')
    Local $x,$end=$a[0][0];EndOf($a)
    For $x=1 To $end
      GE_DoType($a[$x][0])
    Next
  EndFunc
  Func GE_ReadBase()
    Local $a=GE_IniSectionRead('_base')
    Local $x;
    For $x=1 To $a[0][0]
      Local $n=$a[$x][0]
      Local $v=$a[$x][1]
      $v=ToList($v)
      ;DoAssign($n,$v,$k=Default,$force=Default)
      DoAssign('__base_'&$n,$v,'__base');
    Next
  EndFunc
  Func GE_AsWhat($v,$w)
    If GE_IsSectionBase('dim') Then
      Return GE_AsWhatDim($v,$w)
    EndIf
    If GE_IsSectionBase('point') Then
      Return GE_AsWhatPoint($v,$w)
    EndIf
    If GE_IsSectionBase('list') Then
      Return GE_AsWhatList($v,$w)
    EndIf
    If GE_IsSectionBase('point') Then
      Return GE_AsWhatPoint($v,$w)
    EndIf
    Return Call($w&'0',$v)
  EndFunc
  Func GE_AsWhatPoint($v,$w)
    If $v='' Then Return GE_WhatPointArray(0,0)
    If Not StringHas($v,' ') Then
      ErrorExit('Invalid syntax for point:'&$v);
    EndIf
    ;MsgBox(0,'v',$v)
    Local $a=ToList($v);
    Local $x=Call($w&'0',$a[0])
    Local $y=Call($w&'0',$a[1])
    Return GE_WhatPointArray($x,$y)
  EndFunc
  Func GE_AsWhatList($v,$w)
    If $w='Literal' Then
      Return ToList($v)
    Else
      Return ToList(Execute0($v))
    EndIf
  EndFunc
  Func GE_AsWhatDim($v,$w)
    If $v='' Then Return $v
    Local $a=ToList($v)
    Local $c[UBound($a)][2]
    Local $x,$end=EndOf($a)
    For $x=0 To $end
      $v=$a[$x]
      Local $f=StringLeft($v,1);
      Local $n=''
      If GE_IsDim($f) Then
        $v=StringMid($v,2)
        $n=$f;
      EndIf
      $c[$x][0]=$n
      $c[$x][1]=Call($w&'0',$v);
    Next
    Return $c;
  EndFunc
  Func GE_AsWho($n,$t)
    If $t='Many' Then
      Return GE_ManySplit($n)
    EndIf
    Local $a=GE_ManySplit($n)
    Local $x,$end=EndOf($a)
    For $x=0 To $end
      If $t='Lace' Then
        $a[$x]=StringReplace($__section,' ','_'&$a[$x]&'_')
      EndIf
      If $t='Pfx' Then
        $a[$x]=$a[$x]&$__section
      EndIf
    Next
    Return $a;
  EndFunc
  Func GE_ManySplit($s)
    Return ToList(GE_Many($s))
  EndFunc
  Func GE_ManyItem($s)
    Local $c=':';
    Local $d='_';
    If StringInStr($s,'|') Then
      $c='|'
      $d=''
    EndIf
    Local $a=ToList($s,$c);
    $a=ArrayXlat($a,'ToList')
    Return GE_BarZipA($a,$d)
  EndFunc
  Func GE_Many($s)
    Local $a=StringSplit($s,',',2);
    Local $s2=''
    Local $x,$end=EndOf($a)
    For $x=0 To $end
      Local $va=GE_ManyItem($a[$x]);
      $va=_ArrayToString($va,' ')
      $s2=ListAppend($s2,$va,' ')
    Next
    Return $s2
  EndFunc
  Func GE_BarZipA($a,$c=Default)
    Local $p=$a[0]
    Local $x,$end=EndOf($a)
    For $x=1 To $end
      $p=GE_BarZip($p,$a[$x],$c)
    Next
    Return $p
  EndFunc
  Func GE_BarZip($a,$b,$c=Default)
    If $c=Default Then $c='_'
    Local $s='';
    Local $x,$endx=EndOf($a)
    For $x=0 To $endx
      Local $y,$endy=EndOf($b)
      For $y=0 To $endy
        $s=ListAppend($s,$a[$x]&$c&$b[$y],' ')
      Next
    Next
    Return ToList($s)
  EndFunc
  Func GE_IniSectionRead($sec)
    Assign('__section',$sec,2)
    Local $a=IniReadSection($f_in,$sec);
    Return $a
  EndFunc
  Func GE_IsSectionBase($name)
    Return ArrayHas(Eval('__base_'&$name),$__section);
  EndFunc
  Func GE_IsDim($n)
    Return ArrayHas($__base_dims,$n);
  EndFunc
  Func GE_IsSectionQuiet($sec)
    Return Not ArrayHas($__base_required,$sec);
  EndFunc
  Func GE_InduceArrays()
    Local $a=ToList($__assigns);
    Local $x,$end=EndOf($a)
    For $x=0 To $end
      Local $n=$a[$x]
      Local $v=Eval($n)
      If StringInStr($v,' ')<>0 Then
        Local $l=ToList($v)
        DoAssign($n&'_a',$l,AssignKeyOf($n),False)
      EndIf
    Next
  EndFunc
  Func GE_WhatPointArray($x,$y)
    Local $a[2]
    $a[0]=$x;
    $a[1]=$y;
    Local $c[3][2]
    $c[0][0]='a';
    $c[0][1]=$a;
    $c[1][0]='x';
    $c[1][1]=$x;
    $c[2][0]='y';
    $c[2][1]=$y;
    Return $c;
  EndFunc
