  Func DoAssign($n,$v,$k=Default,$force=Default)
    ;$_assigns=$_assigns&@CRLF&$n;
    If $force=Default Then $force=False
    If $k=Default Then $k=False
    If $force Or Not IsDeclared($n) Then
      $__assigns=ListAppendUnique($__assigns,$n,' ')
      Assign($n,$v,2)
      DA_CheckFullfill($n,$k)
    Else
      AssignListValsClip()
      ErrorExit($n&' cannot be set to "'&$v&'" is already defined as "'&Eval($n)&'", section on:'&$__section);
    EndIf
    If $k<>False Then
      If Not IsDeclared($k) Then
        DoAssign($k,'')
      EndIf
      Local $vv=Eval($k)
      Assign($k,ListAppendUnique($vv,$n,' '));
      $__assignskeys=ListAppendUnique($__assignskeys,$n&'='&$k,' ');
    EndIf
  EndFunc

  Func DA_Fullfill($n,$k=Default)
    DA_Fullfill2($n,'l','x',$k)
    DA_Fullfill2($n,'t','y',$k)
    If IsDeclared($n&'_w')==0 And IsDeclared($n&'_x')==1 And IsDeclared($n&'_r')==1 Then DA_AssignFullfill($n&'_w',Eval($n&'_r')-Eval($n&'_x'),$k)
    If IsDeclared($n&'_h')==0 And IsDeclared($n&'_y')==1 And IsDeclared($n&'_b')==1 Then DA_AssignFullfill($n&'_h',Eval($n&'_b')-Eval($n&'_y'),$k)
    If IsDeclared($n&'_r')==0 And IsDeclared($n&'_x')==1 And IsDeclared($n&'_w')==1 Then DA_AssignFullfill($n&'_r',Eval($n&'_x')+Eval($n&'_w'),$k)
    If IsDeclared($n&'_b')==0 And IsDeclared($n&'_y')==1 And IsDeclared($n&'_h')==1 Then DA_AssignFullfill($n&'_b',Eval($n&'_y')+Eval($n&'_h'),$k)
    If IsDeclared($n&'_x')==0 And IsDeclared($n&'_r')==1 And IsDeclared($n&'_w')==1 Then DA_AssignFullfill($n&'_x',Eval($n&'_r')-Eval($n&'_w'),$k)
    If IsDeclared($n&'_y')==0 And IsDeclared($n&'_b')==1 And IsDeclared($n&'_h')==1 Then DA_AssignFullfill($n&'_y',Eval($n&'_b')-Eval($n&'_h'),$k)
    DA_Fullfill2($n,'l','x',$k)
    DA_Fullfill2($n,'t','y',$k)
    If $flag Then
      ClipMsg()
    EndIf
  EndFunc
  Func AssignListShow($n)
    Local $v=''

  Func AssignListVals()
    Local $v=''
    Local $n='__assigns'
    If IsDeclared($n) Then
      $v=Eval($n)
    EndIf
    Local $a=ToList($v)
    Local $s='';
    Local $x;
    Local $end=EndOf($a)
    For $x=0 To $end
      Local $n=$a[$x];
      If $s<>'' Then $s=$s&@CRLF;
      $s=$s&$n&'=';
      $v=Eval($n);
      If IsArray($v) Then $s=$s&'"'&_ArrayToString($v,'","')&'"';
      If IsString($v) Then $s=$s&'"'&$v&'"'
      If IsNumber($v) Then $s=$s&$v
      If IsBool($v) Then $s=$s&$v
    Next
    Return $s
    ;DoSelect_("Assigns "&$n,$s,'')
  EndFunc

  Func DA_IsPoint($k)
    $k=StringMid($k,2)
    If Not IsDeclared('__base_point') Then Return False;
    Return ArrayHas($__base_point,$k);
  EndFunc
  Func DA_CheckFullfill($n,$k)
    If StringInStr($n,'_')=0 Then Return
    If DA_IsPoint($k) Then Return
    Local $l=StringRight($n,2)
    Local $l1=StringLeft($l,1)
    Local $l2=StringRight($l,1)
    If $l1<>'_' Then Return
    If StringInStr('xytlbrhw',$l2)<>0 Then
      If $n='ConsoleFilterBar_y' Then
        ;Msg($n)
        ;$flag=True
      Else
        $flag=False
      EndIf
      DA_Fullfill(StringLeft($n,StringLen($n)-2),$k)
    EndIf
  EndFunc
  Func DA_AssignFullfill($n,$v,$k=Default)
    ;$_assigns=$_assigns&@CRLF&$n;
    If $k=Default Then $k=False
    $__assigns=ListAppendUnique($__assigns,$n,' ')
    Assign($n,$v,2)
    If $k<>False Then
      If Not IsDeclared($k) Then
        DoAssign($k,'')
      EndIf
      Local $vv=Eval($k)
      Assign($k,ListAppendUnique($vv,$n,' '));
    EndIf
  EndFunc
  Func AssignKeyOf($n)
    Local $a=ToList($__assignskeys,' ');
    Local $x,$end=EndOf($a)
    For $x=0 To $end
      Local $i=$a[$x]
      Local $kv=ToList($i,'=')
      If $kv[0]=$n Then
        Return $kv[1]
      EndIf
    Next
    Return ''
  EndFunc
