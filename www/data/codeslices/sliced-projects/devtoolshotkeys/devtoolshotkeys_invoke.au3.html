#include-once

Func DoInvoke()
  DoAssign('name','','__invoke_au3');
  DoAssign('type','','__invoke_au3');
  GetType()
  ;Inform('name type');
EndFunc


Func GetType()
  Local $a=ToList($p1,':')
  Local $t=$a[0];
  Local $n=$a[1];
  If ArrayHas($types_a,$t) Then
    If ArrayHas(Eval($t&'_a'),$n) Then
      $type=$t;
      $name=$n;
    Else
      If StringRight($t,4)='Pane' Then
        If StringLeft($n,6)='Scroll' Then
          $type=$t;
          $name=$n;
        Else
          ErrorExit("GetType:Name for Pane Type is not Scroll:"&$n&" ");
        EndIf
      Else
        ErrorExit("GetType:Cannot find Type:"&$t&" is not Pane");
      EndIf
    EndIf
  Else
    ErrorExit("GetType:Cannot find Type:"&$t);
  EndIf
EndFunc
Func GetTypeWith($n)
  If StringInStr($p1,$n&':')<>0 Then
    $type=$n;
    $name=StringReplace($p1,$n&':','');
    $name=StringReplace($name,'-','');
    Return True
  EndIf
  Return False
EndFunc

Func ValidateType()
  Local $type_idx=_ArraySearch($types_a,$type);
  If $type_idx<>-1 Then
    Return True;Call('Do_'&$type);
  Else
    MsgBox(0,'Type Not Found: Exiting',$type);
    Exit
  EndIf
EndFunc

Func ValidateName()
  Local $test=Eval($type&'_a')
  If IsArray($test) Then
    ;_ArrayDisplay($test)
    Local $i=_ArraySearch($test,$name);
    If $i<>-1 Then
      Return True;
    Else
      MsgBox(0,'Name Not Found for Type '&$type&': Exiting',$name);
      Exit;
    EndIf
  Else
    MsgBox(0,'Cmd Name Array Not Found for Type '&$type&': Exiting',$name);
    Exit
  EndIf
EndFunc
