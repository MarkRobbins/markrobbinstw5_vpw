   _html:function _html__Mn(o,root){
      var me=window.Mn;
      var s='';
      for (i in o) {
        var j=o[i];
        if (j.cmd) {
          if (root) {
            s+='<li>';
          }
          s+='<a';
          s+=' cmd="'+j.cmd+'"';
          if (j.cls) {
            s+=' class="'+j.cls+' '+me.cls.menuitem+'"';
          }else{
            s+=' class="'+me.cls.menuitem+'"';
          }
          s+='>';
          s+=i;
          s+='</a>';
          if (root) {
            s+='</li>';
          }
        }else{//no cmd
          if (root) {
            s+='<li>';
            s+='<a class="'+me.cls.menuparent+'">';
            s+=i;
            s+='</a>';
            s+='<div class="'+me.cls.menucontainer+'">';
            s+=me._html(j,false);
            s+='</div>';
          }else{
            s+='<a';
            if (j.cls) {
              s+=' class="'+j.cls+' '+me.cls.menuitem+'"';
            }else{
              s+=' class="'+me.cls.menuitem+'"';
            }
            s+='>';
            s+=i;
            s+='</a>';
          }
        }
      }
      return s;
     },
