     var msg = "Handler for which:"+event.which+" keyCode"+event.keyCode+" .keypress() called.";

     $('body').keypress(function(event) {
       Kb._bugEvent(event);
       if ( event.which == 13 ) {
          //event.preventDefault();

         return;
       }

         Db.PageUp();
         event.preventDefault();
         return;
       }
       if ( event.keyCode == 34 ) {//pgdn
         Db.PageDown();
         event.preventDefault();
         return;
       }
