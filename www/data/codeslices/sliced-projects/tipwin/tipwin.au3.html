#NoTrayIcon

Global $thisfile="C:\batch\TipWin.au3"
Global $logg="C:\$data\logs\TipWin-log.txt";
Global $timestamp=@YEAR&"."&@MON&"."&@MDAY&"."&@HOUR&"."&@MIN&"."&@SEC&"."&@MSEC

Global $testing=True

If $CmdLine[0]<>0 Then
  $testing=False
EndIf

Global $p1

If Not $testing Then
  If $CmdLine[0]<>1 Then
    RunWait(@AutoItExe&' C:\batch\splash.au3 "Need 1 params - exiting" "'&$thisfile&'" "10"',""); ,@SW_NORMAL)
    Exit
  EndIf
  $p1=$CmdLine[1]
EndIf

;"C:\Program Files (x86)\AutoIt3\AutoIt3.exe" C:\batch\TipWin.au3 E:\$res\kbd\kbd-winkey.png

Local $exe="C:\Program Files (x86)\FullScreen Photo Viewer\FullScreen Photo Viewer.exe"
Local $file='"'&$p1&'"'
; exe <file> [/[a][c][s][3]
;a imgs in subdirs
;c color mgmt
;s hide buttons
;3 3d mode
Opt("WinTitleMatchMode",1)
;1 from start
;2 substr
;3 exact
Local $tn="Fullscreen Image Viewer"

$h1=WinGetHandle($tn)
If $h1<>0 Then
  ;MsgBox(0,$thisfile,"has h1")
EndIf

Run($exe&' "'&$file&'" /s',"");@SW_NORMAL)

Func ts()
  return @YEAR&"."&@MON&"."&@MDAY&"."&@HOUR&"."&@MIN&"."&@SEC&"."&@MSEC
EndFunc
Func logline($line)
  Local $fh1=FileOpen($logg,1);
 If $fh1<>-1 Then
  FileWriteLine($fh1,$line)
  FileClose($fh1)
  EndIf
EndFunc
