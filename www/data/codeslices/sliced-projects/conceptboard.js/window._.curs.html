      ,curs:{
        _styles:function(){
          var s='';
          var NL="\n";
          s+='';
          s+='#curs{box-shadow:0px 0px 2px #337,0px 0px 20px 3px #225;border-radius:10px;position:absolute;z-index:90000;pointer-events:none;}'+NL;
          //s+='.curs{}'+NL;
          return {style:s,name:'curs'};
          } //-_styles
        ,_q:null
        ,_ts:''
        ,_last_pos:null
        ,tgt:null
        ,_get_pos:function(){
          var tg=this.tgt;
          var l=tg.offset().left;
          var t=tg.offset().top;
          l=tg.parent().position().left+tg.position().left;
          t=tg.parent().position().top+tg.position().top+'#padViewport'.q.offset().top;
          var w=tg.width();
          var h=tg.height();
          var pad=10;
          l=l-pad;
          t=t-pad;
          w=w+pad+pad;
          h=h+pad+pad;
          l=''+l+'px';
          t=''+t+'px';
          w=''+w+'px';
          h=''+h+'px';
          var cp={left:l,top:t,width:w,height:h};
          return cp;
          } //-_get_pos
        ,_indicate_at_pos:function(cp){
          this._last_pos=cp;
          this._q.animate(cp,{duration:1000});
          } //-_indicate_at_pos
        ,_indicate:function(){
          var tg=this.tgt;
          if (tg.length!==1) {
            this._q.fadeOut();
            return;
          }
          var cp=this._get_pos();//{left:l,top:t,width:w,height:h};
          var doit=false;
          if (this._last_pos===null) {
            doit=true;
          }else if (cp.left!==this._last_pos.left||cp.top!==this._last_pos.top||cp.width!==this._last_pos.width||cp.height!==this._last_pos.height) {
            doit=true;
          }
          if (!doit) {
            return;
          }
          //this._q.css({left:l,top:t,width:w,height:h});
          this._last_pos=cp;
          if (this._q.is(':visible')) {
            this._q.fadeIn();
          }
          this._q.stop().animate(cp,{duration:1000});
          } //-_indicate
        ,_indicate_cache:function(){
          } //-_indicate_cache
        ,_select:function(q){
          if (this._ts===q.attr('id')) {
            return;
          }
          this._ts=q.attr('id');
          this._indicate();
        }
        ,select:function(q){
          return this._select(q);
        }
        ,deselect:function(){
          this._ts='';
          this.fadeOut();
        }
        ,update:function(){
          this._indicate();
        }
        ,select_dir:function(dir){
          var tgt=this.tgt;
          if (tgt.length===0) {
            return;
          }
          //.var tb=new Textbox(tgt);
          var ltbrs=Ltbr.ltbrs_of_qs('.pWTD-boxStyle'.q);
          delete ltbrs[tgt.attr('id')];
          var ltbr=new Ltbr(tgt);
          var ss=Ltbr.filter_dir(dir,ltbr,ltbrs);
          if (ss) {
            this.select(('#'+ss).q)
          }
          } //-select_dir
        ,in_view:function(){
          var tg=this.tgt;
          if (tg.length!==1) {
            return;
          }
          var tb=new Textbox(tg);
          return tb.in_view();
          } //-in_view
        ,edit_selected:function(){
          var tg=this.tgt;
          if (tg.length!==1) {
            return;
          }
          var tb=new Textbox(tg);
          tb.edit();
          } //-edit_selected
        ,init:function(){// see.update
          var that=this;
          this._q='body'.q.append('<div id="curs"/>').find('#curs');
          'tgt'._ropf(this,function(){
           if (!that._ts) {
             return $([]);
           }
           return ('#'+that._ts).q;
          });
          var q='.pWTD-boxStyle'.q.eq(0);
          if (q.length==1) {
            console.info('indicated');
            this.select(q);
          }
        }
        } //-curs
