      Func EvtError_JsAio()
        Local $oError=$oError__JsAio;
        If Not IsObj($oError) Then
          logerr__JsAio('NoErrror Obj')
          snarl__JsAio(10,'NoErrror Obj','')
          Exit
        EndIf
        Local $line="[COM Error]:s"&$oError.source _
        &" ScriptLine(" & $oError.scriptline & ")" _
        &" : Number 0x" & Hex($oError.number, 8) & " - " & $oError.windescription
        Local $line2="" &$oError.source &':'& $oError.scriptline & ":" & $oError.windescription
        ;Local $xxx=$_com.Error
        Local $xxx=$JsComProxy.Error
        ;$xxx=$jso.Error()
        If $oError.source=='Microsoft JScript runtime error' Then
          Local $line3=''
          $line3&='Source:'&$JsComProxy.Error.Source&@CRLF;
          $line3&='Desc:'&$JsComProxy.Error.Description&@CRLF;
          $line3&='Text:'&$JsComProxy.Error.Text&@CRLF;
          $line3&='Number:'&$JsComProxy.Error.Number&@CRLF;
          $line3&='HelpFile:'&$JsComProxy.Error.HelpFile&@CRLF;
          $line3&='HelpContext:'&$JsComProxy.Error.HelpContext&@CRLF;
          $line3&='Line:'&$JsComProxy.Error.Line&@CRLF;
          $line3&='Column:'&$JsComProxy.Error.Column&@CRLF;
          logerr__JsAio($line3)
          MsgBox(0,'',$line3)
        EndIf
          MsgBox(0,'',$line)
          logerr__JsAio($line)
        Exit
      EndFunc

        Func newJScriptObj__JsAio($VbScriptSupport=False)
          Local $o
          Local $a;
          _FileReadToArray($jsObjCodeFile__JsAio,$a)
          Local $code=_ArrayToString($a,"",1)
          $o = ObjCreate("ScriptControl")
          $o.Language = "JScript"
          $o.Timeout = 60000;
          $o.AddCode($code)
          If $VbScriptSupport = True Then $o.Run("set_vbscript", ObjCreate("ScriptControl"))
          Return $o
        EndFunc
